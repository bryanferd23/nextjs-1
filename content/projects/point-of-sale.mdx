# Point of Sale System

A comprehensive web-based Point of Sale (POS) system built with vanilla JavaScript, HTML, and Tailwind CSS that provides retail businesses with essential tools for transaction processing, inventory management, and sales analytics. The system is designed to be lightweight, fast, and user-friendly for small to medium-sized retail operations.

## Project Overview

This Point of Sale system was developed to provide small and medium-sized retail businesses with an affordable, browser-based solution for managing daily sales operations. The system eliminates the need for expensive hardware while offering essential POS functionality through a modern, responsive web interface.

## Key Features

### 💳 Transaction Processing
- Fast and intuitive sales interface
- Multiple payment method support (cash, card, digital)
- Real-time transaction calculations with tax

- Transaction history and search functionality

### 📦 Inventory Management

```javascript
                        this.startNewTransaction();

                    case 'p': // Process payment
                        e.preventDefault();
                        this.processPayment();
                        break;
                    case 'r': // Reports
                        e.preventDefault();
                        this.showReports();
                        break;
                }

        document.getElementById('product-search').addEventListener('input', (e) => {
            this.searchProducts(e.target.value);

            if (e.target.classList.contains('remove-item')) {
                this.cart.removeItem(e.target.dataset.productId);

        });
```

### Shopping Cart Management
```javascript
// Advanced shopping cart with tax calculations and discounts
class ShoppingCart {
    constructor() {
        this.items = [];
        this.discounts = [];
        this.taxRate = 0.12; // 12% VAT
        this.customer = null;
    }

    /**
     * Add product to cart with quantity and price validation
     */
    addItem(product, quantity = 1, customPrice = null) {
        if (!this.validateProduct(product)) {
            throw new Error('Invalid product data');
        }

        if (quantity <= 0) {
            throw new Error('Quantity must be greater than 0');
        }

        // Check stock availability
        if (product.stock < quantity) {
            throw new Error(`Insufficient stock. Available: ${product.stock}`);
        }

        const existingItem = this.items.find(item => item.product.id === product.id);
        
        if (existingItem) {
            const newQuantity = existingItem.quantity + quantity;
            if (newQuantity > product.stock) {
                throw new Error(`Total quantity exceeds available stock`);
            }
            existingItem.quantity = newQuantity;
        } else {
            this.items.push({
                product: { ...product },
                quantity,
                unitPrice: customPrice || product.price,
                discount: 0,
                id: this.generateItemId()
            });
        }

        this.updateTotals();
        this.triggerCartUpdate();
    }

    /**
     * Apply discount to specific item or entire cart
     */
    applyDiscount(discountData) {
        if (discountData.type === 'item' && discountData.itemId) {
            const item = this.items.find(i => i.id === discountData.itemId);
            if (item) {
                item.discount = this.calculateDiscountAmount(
                    item.unitPrice * item.quantity,
                    discountData
                );
            }
        } else if (discountData.type === 'cart') {
            this.discounts.push(discountData);
        }

        this.updateTotals();
    }

    /**
     * Calculate comprehensive cart totals
     */
    calculateTotals() {
        const subtotal = this.items.reduce((total, item) => {
            return total + (item.unitPrice * item.quantity) - item.discount;
        }, 0);

        const cartDiscountAmount = this.discounts.reduce((total, discount) => {
            return total + this.calculateDiscountAmount(subtotal, discount);
        }, 0);

        const discountedSubtotal = subtotal - cartDiscountAmount;
        const taxAmount = discountedSubtotal * this.taxRate;
        const total = discountedSubtotal + taxAmount;

        return {
            subtotal: subtotal,
            discounts: cartDiscountAmount,
            discountedSubtotal: discountedSubtotal,
            tax: taxAmount,
            total: total,
            itemCount: this.items.reduce((count, item) => count + item.quantity, 0)
        };
    }

    /**
     * Generate receipt data for printing
     */
    generateReceiptData() {
        const totals = this.calculateTotals();
        const receiptNumber = this.generateReceiptNumber();
        
        return {
            receiptNumber,
            timestamp: new Date(),
            cashier: this.getCurrentCashier(),
            customer: this.customer,
            items: this.items.map(item => ({
                name: item.product.name,
                quantity: item.quantity,
                unitPrice: item.unitPrice,
                total: (item.unitPrice * item.quantity) - item.discount,
                discount: item.discount
            })),
            totals,
            paymentMethods: []
        };
    }
}
```

```javascript
// Comprehensive inventory management
class InventoryManager {
    constructor() {
        this.products = new Map();
        this.categories = new Map();
        this.lowStockThreshold = 10;
    }

    /**
     * Load products from local storage or API
     */
    async loadProducts() {
        try {
            const storedProducts = localStorage.getItem('pos_products');
            if (storedProducts) {
                const products = JSON.parse(storedProducts);
                products.forEach(product => {
                    this.products.set(product.id, product);
                });
            } else {
                // Load sample data if no products exist
                await this.loadSampleData();
            }
            
            this.updateLowStockAlerts();
        } catch (error) {
            console.error('Failed to load products:', error);
            throw new Error('Could not load product inventory');
        }
    }

    /**
     * Add or update product in inventory
     */
    addOrUpdateProduct(productData) {
        const product = {
            id: productData.id || this.generateProductId(),
            name: productData.name,
            description: productData.description || '',
            category: productData.category,
            price: parseFloat(productData.price),
            cost: parseFloat(productData.cost) || 0,
            stock: parseInt(productData.stock) || 0,
            minStock: parseInt(productData.minStock) || this.lowStockThreshold,
            barcode: productData.barcode || '',
            sku: productData.sku || this.generateSKU(),
            image: productData.image || 'default-product.jpg',
            isActive: productData.isActive !== false,
            createdAt: productData.createdAt || new Date(),
            updatedAt: new Date()
        };

        this.products.set(product.id, product);
        this.saveToStorage();
        this.updateLowStockAlerts();
        
        return product;
    }

    /**
     * Search products with advanced filtering
     */
    searchProducts(query, filters = {}) {
        const searchResults = Array.from(this.products.values()).filter(product => {
            // Text search
            const matchesQuery = !query || 
                product.name.toLowerCase().includes(query.toLowerCase()) ||
                product.description.toLowerCase().includes(query.toLowerCase()) ||
                product.barcode.includes(query) ||
                product.sku.includes(query);

            // Category filter
            const matchesCategory = !filters.category || 
                product.category === filters.category;

            // Price range filter
            const matchesPrice = (!filters.minPrice || product.price >= filters.minPrice) &&
                               (!filters.maxPrice || product.price <= filters.maxPrice);

            // Stock filter
            const matchesStock = !filters.stockFilter || 
                (filters.stockFilter === 'low' && product.stock <= product.minStock) ||
                (filters.stockFilter === 'out' && product.stock === 0) ||
                (filters.stockFilter === 'available' && product.stock > 0);

            return matchesQuery && matchesCategory && matchesPrice && matchesStock && product.isActive;
        });

        // Sort results
        return searchResults.sort((a, b) => {
            switch (filters.sortBy) {
                case 'name':
                    return a.name.localeCompare(b.name);
                case 'price':
                    return a.price - b.price;
                case 'stock':
                    return a.stock - b.stock;
                default:
                    return a.name.localeCompare(b.name);
            }
        });
    }

    /**
     * Update stock levels after sale
     */
    updateStock(productId, quantitySold) {
        const product = this.products.get(productId);
        if (!product) {
            throw new Error('Product not found');
        }

        if (product.stock < quantitySold) {
            throw new Error('Insufficient stock');
        }

        product.stock -= quantitySold;
        product.updatedAt = new Date();
        
        this.saveToStorage();
        this.updateLowStockAlerts();

        // Log inventory movement
        this.logInventoryMovement(productId, -quantitySold, 'sale');
    }

    /**
     * Generate comprehensive inventory report
     */
    generateInventoryReport() {
        const products = Array.from(this.products.values());
        
        const report = {
            totalProducts: products.length,
            activeProducts: products.filter(p => p.isActive).length,
            totalValue: products.reduce((total, p) => total + (p.price * p.stock), 0),
            totalCost: products.reduce((total, p) => total + (p.cost * p.stock), 0),
            lowStockItems: products.filter(p => p.stock <= p.minStock && p.isActive),
            outOfStockItems: products.filter(p => p.stock === 0 && p.isActive),
            topValueItems: products
                .filter(p => p.isActive)
                .sort((a, b) => (b.price * b.stock) - (a.price * a.stock))
                .slice(0, 10),
            categoryBreakdown: this.getCategoryBreakdown(products)
        };

        return report;
    }
}
```

```javascript
// Advanced reporting and analytics system
class ReportsManager {
    constructor() {
        this.transactions = [];
        this.reportCache = new Map();
    }

    /**
     * Generate comprehensive sales report
     */
    generateSalesReport(startDate, endDate, filters = {}) {
        const cacheKey = `sales_${startDate}_${endDate}_${JSON.stringify(filters)}`;
        
        if (this.reportCache.has(cacheKey)) {
            return this.reportCache.get(cacheKey);
        }

        const filteredTransactions = this.filterTransactions(startDate, endDate, filters);
        
        const report = {
            period: { startDate, endDate },
            summary: this.calculateSummary(filteredTransactions),
            dailyBreakdown: this.getDailyBreakdown(filteredTransactions),
            hourlyBreakdown: this.getHourlyBreakdown(filteredTransactions),
            topProducts: this.getTopProducts(filteredTransactions),
            paymentMethods: this.getPaymentMethodBreakdown(filteredTransactions),
            cashierPerformance: this.getCashierPerformance(filteredTransactions),
            customerAnalysis: this.getCustomerAnalysis(filteredTransactions)
        };

        this.reportCache.set(cacheKey, report);
        return report;
    }

    /**
     * Calculate key performance indicators
     */
    calculateSummary(transactions) {
        const totalSales = transactions.reduce((sum, t) => sum + t.total, 0);
        const totalTransactions = transactions.length;
        const totalItems = transactions.reduce((sum, t) => 
            sum + t.items.reduce((itemSum, item) => itemSum + item.quantity, 0), 0
        );

        const averageTransaction = totalTransactions > 0 ? totalSales / totalTransactions : 0;
        const averageItems = totalTransactions > 0 ? totalItems / totalTransactions : 0;

        return {
            totalSales,
            totalTransactions,
            totalItems,
            averageTransaction,
            averageItems,
            totalDiscounts: transactions.reduce((sum, t) => sum + (t.discount || 0), 0),
            totalTax: transactions.reduce((sum, t) => sum + (t.tax || 0), 0)
        };
    }

    /**
     * Generate real-time dashboard data
     */
    generateDashboardData() {
        const today = new Date();
        const todayStart = new Date(today.getFullYear(), today.getMonth(), today.getDate());
        const todayTransactions = this.transactions.filter(t => 
            new Date(t.timestamp) >= todayStart
        );

        const thisWeekStart = new Date(today);
        thisWeekStart.setDate(today.getDate() - today.getDay());
        const weekTransactions = this.transactions.filter(t => 
            new Date(t.timestamp) >= thisWeekStart
        );

        const thisMonthStart = new Date(today.getFullYear(), today.getMonth(), 1);
        const monthTransactions = this.transactions.filter(t => 
            new Date(t.timestamp) >= thisMonthStart
        );

        return {
            today: this.calculateSummary(todayTransactions),
            week: this.calculateSummary(weekTransactions),
            month: this.calculateSummary(monthTransactions),
            recentTransactions: this.transactions
                .sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp))
                .slice(0, 10),
            lowStockAlerts: this.getLowStockAlerts(),
            topSellingToday: this.getTopProducts(todayTransactions, 5)
        };
    }
}
```

```javascript
// Web-based barcode scanner using camera API
class BarcodeScanner {
    constructor() {
        this.isScanning = false;
        this.stream = null;
    }

    async startScanning() {
        if (this.isScanning) return;

        try {
            this.stream = await navigator.mediaDevices.getUserMedia({
                video: { 
                    facingMode: 'environment',
                    width: { ideal: 1280 },
                    height: { ideal: 720 }
                }
            });

            const video = document.getElementById('barcode-video');
            video.srcObject = this.stream;
            
            this.isScanning = true;
            this.scanLoop();
        } catch (error) {
            console.error('Failed to start barcode scanner:', error);
            throw new Error('Camera access denied or not available');
        }
    }

    scanLoop() {
        if (!this.isScanning) return;

        const canvas = document.getElementById('barcode-canvas');
        const video = document.getElementById('barcode-video');
        const ctx = canvas.getContext('2d');

        canvas.width = video.videoWidth;
        canvas.height = video.videoHeight;
        ctx.drawImage(video, 0, 0);

        const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
        const barcode = this.detectBarcode(imageData);

        if (barcode) {
            this.handleBarcodeDetected(barcode);
            this.stopScanning();
        } else {
            requestAnimationFrame(() => this.scanLoop());
        }
    }
}
```
