# Boarding House Management System (BHMS)

A comprehensive property management solution built with Laravel that streamlines boarding house operations, from tenant management and room allocation to payment processing and maintenance tracking. The system is designed to handle the complex needs of boarding house operators while providing an intuitive experience for both administrators and tenants.

## Project Overview

The Boarding House Management System was developed to address the unique challenges faced by boarding house operators who need to manage multiple tenants, rooms, payments, utilities, and maintenance tasks efficiently. The system provides a centralized platform that automates routine tasks and provides valuable insights into property performance.

```php
<?php
// Property Controller - Main property management logic
namespace App\Http\Controllers;

use App\Http\Controllers\Controller;
use App\Http\Requests\StorePropertyRequest;
use App\Http\Requests\UpdatePropertyRequest;
use App\Models\Property;
use App\Models\Room;
use App\Services\PropertyService;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;

class PropertyController extends Controller
{
    protected PropertyService $propertyService;

    public function __construct(PropertyService $propertyService)
    {
        $this->propertyService = $propertyService;
        $this->middleware('auth');
        $this->middleware('role:admin,manager')->except(['show', 'index']);
    }

    /**
     * Display properties with advanced filtering and analytics
     */
    public function index(Request $request)
    {
        $filters = $request->only(['status', 'type', 'location', 'occupancy_rate']);
        $sortBy = $request->get('sort_by', 'name');
        $sortOrder = $request->get('sort_order', 'asc');

        $properties = $this->propertyService->getFilteredProperties(
            $filters,
            $sortBy,
            $sortOrder,
            $request->get('per_page', 15)
        );

        // Calculate analytics for dashboard
        $analytics = $this->propertyService->getPropertiesAnalytics();

        return view('properties.index', compact('properties', 'analytics', 'filters'));
    }

    /**
     * Store a new property with room setup
     */
    public function store(StorePropertyRequest $request)
    {
        try {
            DB::transaction(function () use ($request) {
                $property = $this->propertyService->createProperty($request->validated());
                
                // Create rooms if provided
                if ($request->has('rooms')) {
                    $this->propertyService->createRooms($property, $request->get('rooms'));
                }
                
                // Setup default utilities and amenities
                $this->propertyService->setupDefaultUtilities($property);
                
                return $property;
            });

            return redirect()->route('properties.index')
                           ->with('success', 'Property created successfully');
        } catch (\Exception $e) {
            return back()->withInput()
                        ->with('error', 'Failed to create property: ' . $e->getMessage());
        }
    }

    /**
     * Generate comprehensive property report
     */
    public function generateReport(Property $property, Request $request)
    {
        $reportType = $request->get('type', 'monthly');
        $startDate = $request->get('start_date', now()->startOfMonth());
        $endDate = $request->get('end_date', now()->endOfMonth());

        $report = $this->propertyService->generatePropertyReport(
            $property,
            $reportType,
            $startDate,
            $endDate
        );

        return view('properties.reports.show', compact('property', 'report'));
    }
}
```

```php
<?php
// Property Service - Business logic implementation
namespace App\Services;

use App\Models\Property;
use App\Models\Room;
use App\Models\Tenant;
use App\Models\Payment;
use App\Repositories\PropertyRepository;
use Illuminate\Support\Facades\Cache;
use Illuminate\Support\Facades\Storage;

class PropertyService
{
    protected PropertyRepository $propertyRepository;

    public function __construct(PropertyRepository $propertyRepository)
    {
        $this->propertyRepository = $propertyRepository;
    }

    /**
     * Create property with comprehensive setup
     */
    public function createProperty(array $data): Property
    {
        $property = $this->propertyRepository->create($data);
        
        // Upload property images
        if (isset($data['images'])) {
            $this->uploadPropertyImages($property, $data['images']);
        }
        
        // Generate property QR code for tenant access
        $this->generatePropertyQRCode($property);
        
        // Setup default configuration
        $this->setupDefaultConfiguration($property);
        
        // Clear cache
        $this->clearPropertyCache();
        
        return $property;
    }

    /**
     * Calculate comprehensive property analytics
     */
    public function getPropertiesAnalytics(): array
    {
        return Cache::remember('properties_analytics', 3600, function () {
            $totalProperties = Property::count();
            $totalRooms = Room::count();
            $occupiedRooms = Room::where('status', 'occupied')->count();
            $availableRooms = Room::where('status', 'available')->count();
            
            $occupancyRate = $totalRooms > 0 ? ($occupiedRooms / $totalRooms) * 100 : 0;
            
            $monthlyRevenue = Payment::whereMonth('created_at', now()->month)
                                   ->whereYear('created_at', now()->year)
                                   ->where('status', 'completed')
                                   ->sum('amount');
            
            $pendingPayments = Payment::where('status', 'pending')
                                    ->whereDate('due_date', '<=', now())
                                    ->sum('amount');
            
            return [
                'total_properties' => $totalProperties,
                'total_rooms' => $totalRooms,
                'occupied_rooms' => $occupiedRooms,
                'available_rooms' => $availableRooms,
                'occupancy_rate' => round($occupancyRate, 2),
                'monthly_revenue' => $monthlyRevenue,
                'pending_payments' => $pendingPayments,
                'average_room_rate' => Room::avg('monthly_rate'),
                'maintenance_requests' => $this->getMaintenanceStats(),
                'tenant_satisfaction' => $this->getTenantSatisfactionScore()
            ];
        });
    }

    /**
     * Advanced room allocation algorithm
     */
    public function allocateRoom(Tenant $tenant, array $preferences = []): ?Room
    {
        $query = Room::where('status', 'available')
                    ->where('property_id', $tenant->preferred_property_id);
        
        // Apply preferences
        if (isset($preferences['room_type'])) {
            $query->where('room_type', $preferences['room_type']);
        }
        
        if (isset($preferences['max_budget'])) {
            $query->where('monthly_rate', '<=', $preferences['max_budget']);
        }
        
        if (isset($preferences['floor_preference'])) {
            $query->where('floor', $preferences['floor_preference']);
        }
        
        // Prioritize by tenant score and room quality
        $availableRooms = $query->get();
        
        if ($availableRooms->isEmpty()) {
            return null;
        }
        
        // Score-based allocation
        $bestRoom = null;
        $bestScore = 0;
        
        foreach ($availableRooms as $room) {
            $score = $this->calculateRoomScore($room, $tenant, $preferences);
            
            if ($score > $bestScore) {
                $bestScore = $score;
                $bestRoom = $room;
            }
        }
        
        return $bestRoom;
    }

    /**
     * Generate comprehensive property report
     */
    public function generatePropertyReport(Property $property, string $type, $startDate, $endDate): array
    {
        $report = [
            'property' => $property,
            'period' => compact('startDate', 'endDate', 'type'),
            'occupancy' => $this->getOccupancyReport($property, $startDate, $endDate),
            'financial' => $this->getFinancialReport($property, $startDate, $endDate),
            'maintenance' => $this->getMaintenanceReport($property, $startDate, $endDate),
            'tenant_turnover' => $this->getTenantTurnoverReport($property, $startDate, $endDate)
        ];
        
        // Generate charts data for frontend
        $report['charts'] = $this->generateChartsData($report);
        
        return $report;
    }
}
```

```php
<?php
// Tenant Model with advanced relationships and scopes
namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\BelongsTo;
use Illuminate\Database\Eloquent\Relations\HasMany;
use Illuminate\Database\Eloquent\SoftDeletes;

class Tenant extends Model
{
    use HasFactory, SoftDeletes;

    protected $fillable = [
        'first_name',
        'last_name',
        'email',
        'phone',
        'date_of_birth',
        'occupation',
        'emergency_contact_name',
        'emergency_contact_phone',
        'room_id',
        'lease_start_date',
        'lease_end_date',
        'monthly_rent',
        'security_deposit',
        'status',
        'notes'
    ];

    protected $casts = [
        'date_of_birth' => 'date',
        'lease_start_date' => 'date',
        'lease_end_date' => 'date',
        'monthly_rent' => 'decimal:2',
        'security_deposit' => 'decimal:2'
    ];

    /**
     * Relationships
     */
    public function room(): BelongsTo
    {
        return $this->belongsTo(Room::class);
    }

    public function property(): BelongsTo
    {
        return $this->hasOneThrough(Property::class, Room::class, 'id', 'id', 'room_id', 'property_id');
    }

    public function payments(): HasMany
    {
        return $this->hasMany(Payment::class);
    }

    public function maintenanceRequests(): HasMany
    {
        return $this->hasMany(MaintenanceRequest::class);
    }

    public function applications(): HasMany
    {
        return $this->hasMany(TenantApplication::class);
    }

    /**
     * Query scopes
     */
    public function scopeActive($query)
    {
        return $query->where('status', 'active');
    }

    public function scopeByProperty($query, $propertyId)
    {
        return $query->whereHas('room', function ($q) use ($propertyId) {
            $q->where('property_id', $propertyId);
        });
    }

    public function scopeLeaseExpiring($query, $days = 30)
    {
        return $query->where('lease_end_date', '<=', now()->addDays($days))
                    ->where('status', 'active');
    }

    /**
     * Calculate tenant score for room allocation
     */
    public function calculateTenantScore(): float
    {
        $score = 0;
        
        // Payment history (40% weight)
        $paymentScore = $this->getPaymentHistoryScore();
        $score += $paymentScore * 0.4;
        
        // Length of stay (20% weight)
        $stayScore = $this->getLengthOfStayScore();
        $score += $stayScore * 0.2;
        
        // Maintenance requests (20% weight)
        $maintenanceScore = $this->getMaintenanceScore();
        $score += $maintenanceScore * 0.2;
        
        // References and background (20% weight)
        $backgroundScore = $this->getBackgroundScore();
        $score += $backgroundScore * 0.2;
        
        return round($score, 2);
    }

    /**
     * Get payment history score
     */
    protected function getPaymentHistoryScore(): float
    {
        $totalPayments = $this->payments()->count();
        
        if ($totalPayments === 0) return 5.0; // Neutral score for new tenants
        
        $onTimePayments = $this->payments()
                              ->whereRaw('paid_date <= due_date')
                              ->count();
        
        $latePayments = $this->payments()
                            ->whereRaw('paid_date > due_date')
                            ->count();
        
        $onTimeRate = $onTimePayments / $totalPayments;
        
        // Score from 0-10 based on on-time payment rate
        return $onTimeRate * 10;
    }

    /**
     * Generate lease agreement
     */
    public function generateLeaseAgreement(): string
    {
        $template = Storage::get('templates/lease_agreement.html');
        
        $variables = [
            'tenant_name' => $this->full_name,
            'tenant_email' => $this->email,
            'tenant_phone' => $this->phone,
            'room_number' => $this->room->room_number,
            'property_address' => $this->room->property->full_address,
            'monthly_rent' => number_format($this->monthly_rent, 2),
            'security_deposit' => number_format($this->security_deposit, 2),
            'lease_start_date' => $this->lease_start_date->format('F j, Y'),
            'lease_end_date' => $this->lease_end_date->format('F j, Y'),
            'generated_date' => now()->format('F j, Y')
        ];
        
        foreach ($variables as $key => $value) {
            $template = str_replace("{{$key}}", $value, $template);
        }
        
        return $template;
    }
}
```

### Payment Processing System
```php
<?php
// Payment Service with advanced processing capabilities
namespace App\Services;

use App\Models\Payment;
use App\Models\Tenant;
use App\Notifications\PaymentReminder;
use App\Notifications\PaymentReceived;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Notification;

class PaymentService
{
    /**
     * Process monthly rent payments for all active tenants
     */
    public function processMonthlyRents(): array
    {
        $results = [];
        $tenants = Tenant::active()->with('room')->get();
        
        foreach ($tenants as $tenant) {
            try {
                $payment = $this->createMonthlyRentPayment($tenant);
                $results['created'][] = $payment;
                
                // Send payment notification
                $tenant->notify(new PaymentReminder($payment));
                
            } catch (\Exception $e) {
                $results['errors'][] = [
                    'tenant_id' => $tenant->id,
                    'error' => $e->getMessage()
                ];
            }
        }
        
        return $results;
    }

    /**
     * Create monthly rent payment
     */
    protected function createMonthlyRentPayment(Tenant $tenant): Payment
    {
        $dueDate = now()->addDays(5); // 5 days from now
        
        // Check if payment already exists for this month
        $existingPayment = Payment::where('tenant_id', $tenant->id)
                                 ->where('type', 'rent')
                                 ->whereMonth('due_date', $dueDate->month)
                                 ->whereYear('due_date', $dueDate->year)
                                 ->first();
        
        if ($existingPayment) {
            throw new \Exception('Payment already exists for this month');
        }
        
        return Payment::create([
            'tenant_id' => $tenant->id,
            'type' => 'rent',
            'amount' => $tenant->monthly_rent,
            'due_date' => $dueDate,
            'status' => 'pending',
            'description' => "Monthly rent for {$tenant->room->room_number}",
            'late_fee_amount' => $this->calculateLateFee($tenant->monthly_rent),
            'late_fee_applies_after' => $dueDate->addDays(5)
        ]);
    }

    /**
     * Process payment with multiple gateway support
     */
    public function processPayment(Payment $payment, array $paymentData): bool
    {
        return DB::transaction(function () use ($payment, $paymentData) {
            $gateway = $this->getPaymentGateway($paymentData['method']);
            
            try {
                $result = $gateway->charge([
                    'amount' => $payment->total_amount,
                    'currency' => 'PHP',
                    'description' => $payment->description,
                    'metadata' => [
                        'payment_id' => $payment->id,
                        'tenant_id' => $payment->tenant_id
                    ]
                ] + $paymentData);
                
                if ($result['success']) {
                    $payment->update([
                        'status' => 'completed',
                        'paid_date' => now(),
                        'payment_method' => $paymentData['method'],
                        'transaction_id' => $result['transaction_id'],
                        'gateway_response' => $result['response']
                    ]);
                    
                    // Send confirmation notification
                    $payment->tenant->notify(new PaymentReceived($payment));
                    
                    return true;
                }
                
                throw new \Exception($result['error'] ?? 'Payment failed');
                
            } catch (\Exception $e) {
                $payment->update([
                    'status' => 'failed',
                    'error_message' => $e->getMessage()
                ]);
                
                throw $e;
            }
        });
    }

    /**
     * Generate financial reports
     */
    public function generateFinancialReport($propertyId = null, $startDate = null, $endDate = null): array
    {
        $startDate = $startDate ?: now()->startOfMonth();
        $endDate = $endDate ?: now()->endOfMonth();
        
        $query = Payment::whereBetween('created_at', [$startDate, $endDate]);
        
        if ($propertyId) {
            $query->whereHas('tenant.room', function ($q) use ($propertyId) {
                $q->where('property_id', $propertyId);
            });
        }
        
        $payments = $query->with('tenant.room.property')->get();
        
        return [
            'total_revenue' => $payments->where('status', 'completed')->sum('amount'),
            'pending_payments' => $payments->where('status', 'pending')->sum('amount'),
            'overdue_payments' => $payments->where('status', 'pending')
                                          ->where('due_date', '<', now())->sum('amount'),
            'late_fees_collected' => $payments->sum('late_fee_charged'),
            'payment_methods' => $payments->groupBy('payment_method')
                                        ->map->sum('amount'),
            'monthly_breakdown' => $this->getMonthlyBreakdown($payments),
            'property_breakdown' => $this->getPropertyBreakdown($payments)
        ];
    }
}
```

## Implementation Highlights

### Challenge 1: Complex Financial Calculations
**Problem**: Managing multiple rent types, utilities, late fees, and prorated amounts with different billing cycles.

**Solution**:
- Created flexible payment calculation engine
